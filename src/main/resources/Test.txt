        public String getName() {
            return name;
        }

        @Check
        public void setAge(int a) {
            age = a;
        }
    }

    @Value("name = {test}")
    private final static String mapping;

    @Override
    public void readFile(List<String> file) throws IOException {
        String string = Files.readString(Path.of(file));
        mapping.valueOf(string);
    }
}
